cmake_minimum_required(VERSION 4.1)

project(softRenderer)

set(HOME ${CMAKE_CURRENT_SOURCE_DIR}/home)
set(EXECUTABLE_OUTPUT_PATH ${HOME}/bin)
set(THIRD_PARTY ${CMAKE_CURRENT_SOURCE_DIR}/thirdParty)
file(GLOB SRC_LIST ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

include_directories(${THIRD_PARTY}/inc)
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/inc)

link_directories(${THIRD_PARTY}/lib)

add_subdirectory(SDL)

add_executable(sRenderer ${SRC_LIST})

target_link_options(sRenderer PRIVATE
			      -static 
			      -static-libstdc++
			      -static-libgcc
			      -pthread
			      -Wl,--as-needed
			      )

target_link_libraries(sRenderer PRIVATE SDL3::SDL3-static)
